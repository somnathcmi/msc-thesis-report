\subsection{Notations}
We shall use the following notation.
\begin{itemize}
\item $n$ is number of variables in circuit.
\item $s$ is size of circuit.
\item $r$ is degree of circuit.
\item $N={n+r-1 \choose r}$ is number of $n$-variate monomials of degree $r$.
\item $\bm{f}$ is vector of coefficients of polynomial $f$.
\item $G_\delta=\{-1,-1+\delta,-1+2\delta,...,1-2\delta,1-\delta\}^n$ where $0<\delta<1$. We call it $\delta$-grid in $\IR^n$.
\item $G_\delta^\IC=\{\VEC a+\iota\VEC b:a,b\in G_\delta\}$. We call it $\delta$-grid in $\IC^n$.
\item $[-1,1]^N_\IC=[-1,1]^N+\iota\cdot[-1,1]^N=\{\VEC a+\iota\VEC b: \VEC a,\VEC b \in [-1,1]^N\}$
\end{itemize}
}
